     \else 
       \msiopentag{#1}{<#1>}%
     \fi
     {\everypar={}\msitag{^^0a}%
      \def\short{##1}%

      \ifx\short\@empty
      \else
        \msiopentag{shortTitle}{<shortTitle>}%
          ##1%
        \msiclosetag{shortTitle}{</shortTitle>}%
      \fi
      ##2%
      \msiclosetag{sectiontitle}{</sectiontitle>}%
      \msitag{^^0a}%
     }%
     \par
  }%
}

\def\@inbrace#1{%
   \ifnum\msi@inpreamble=1

  \begingroup
     \msi@inpreamble=0
     \ifx\atok\@bracetok
       \global\let\@donext\@inbrace
     \else

       \else
         \ifx\atok\@startok
           \global\let\@donext\@astar
         \else
           \global\let\@donext\@endgobble
         \fi
       \fi
     \fi
   \endgroup
   \@donext}

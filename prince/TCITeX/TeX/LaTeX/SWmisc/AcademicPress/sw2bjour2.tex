\def\xappendix#1 #2\par{\appendix{#2}}
\def\xappendixtitle#1 #2\par{\appendixtitle{#2}}

\let\qed=\QED

\def\QTR#1{\csname #1\endcsname}

\def\eatone#1{}

\long\def\xtablenote#1\end{\endgroup\tablenotes{#1}\eatone}

\def\lettercaption{\futurelet\next\lookforletter}
\def\lookforletter{\ifx\next[\long\def\go[##1]##2\end##3{\endgroup\letteredcaption{##1}{##2}}
\else\long\def\go##1\end##2{\endgroup\letteredcaption{you should put a letter here}{##1}}\fi\go}

\def\lettercontinuedcaption{\futurelet\next\lookforcletter}
\def\lookforcletter{\ifx\next[\long\def\go[##1]##2\end##3{\endgroup\letteredcontinuedcaption{##1}{##2}}
\else\long\def\go##1\end##2{\endgroup\letteredcontinuedcaption{you should put a letter here}{##1}}\fi\go}

\renewenvironment{tabular}[1]{\begin{tabular*}{\hsize}{@{\extracolsep{\fill}}#1}}{\end{tabular*}}
